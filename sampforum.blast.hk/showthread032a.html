if(PlayerInfo[playerid][Level] == 1338)
    {
    new Text3D:Management3D;
    new Float:X, Float:Y, Float:Z;
    GetPlayerPos(playerid, X, Y, Z);
    Management3D = Create3DTextLabel("Management", MANAGEMENT, X, Y, Z, 40.0, -1, 0);

    }
