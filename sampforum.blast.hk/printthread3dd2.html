pawn Code:
/* --- numbers --- */
stock SizedRandomNumbers(size)
{
    new str[128];
    for(new i=0; i<size; i++)
    {
        format(str, sizeof(str), "%s%d", str, random(9));
    }
    return str;
}

/* --- Characters --- */
